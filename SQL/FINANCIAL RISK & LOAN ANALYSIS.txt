FINANCIAL RISK & LOAN ANALYSIS
# DATABASE CREATION
CREATE DATABASE LOANDB;
USE LOANDB;

# LOAD AND OVERVIEW THE DATA SET
SELECT * FROM loan_data;
EXEC sp_rename 'loan_data','LOAN_DATA';
SELECT * FROM LOAN_DATA;

 # CJECKING NULL VALUES
SELECT COUNT(*) AS TOTAL_ROWS,
 SUM (CASE WHEN Loan_ID IS NULL THEN 1 ELSE 0 END) AS Loan_ID_NULLS,
 SUM (CASE WHEN Gender IS NULL THEN 1 ELSE 0 END) AS Gender_NULLS,
 SUM (CASE WHEN Married IS NULL THEN 1 ELSE 0 END) AS Married_NULLS,
 SUM (CASE WHEN Dependents IS NULL  THEN 1 ELSE 0 END) AS Dependents_NULLS,
 SUM (CASE WHEN Education IS NULL THEN 1 ELSE 0 END) AS Education_NULLS,
 SUM (CASE WHEN Self_Employed IS NULL THEN 1 ELSE 0 END) AS Self_Employed_NULLS,
 SUM (CASE WHEN ApplicantIncome IS NULL THEN 1 ELSE 0 END) AS ApplicantIncome_NULLS,
 SUM (CASE WHEN CoapplicantIncome IS NULL THEN 1 ELSE 0 END) AS CoapplicantIncome_NULLS,
 SUM (CASE WHEN LoanAmount IS NULL THEN 1 ELSE 0 END) AS LoanAmount_NULLS,
 SUM (CASE WHEN Loan_Amount_Term IS NULL THEN 1 ELSE 0 END) AS Loan_Amount_Term_NULLS,
 SUM (CASE WHEN Credit_History IS NULL THEN 1 ELSE 0 END) AS Credit_History_NULLS,
 SUM (CASE WHEN Property_Area IS NULL THEN 1 ELSE 0 END) AS Property_Area_NULLS,
 SUM (CASE WHEN Loan_Status IS NULL THEN 1 ELSE 0 END) AS Loan_Status_NULLS
 FROM LOAN_DATA;

# UPDATE DATA FORMAT
UPDATE LOAN_DATA
SET Dependents = '3'
WHERE Dependents = '3+';

#  CLEAN DATASET 
UPDATE LOAN_DATA
 SET Gender = COALESCE(Gender,'Male'),
  Dependents = COALESCE(Dependents,0),
  Self_Employed = COALESCE(Self_Employed,'NO'),
  Loan_Amount_Term = COALESCE(Loan_Amount_Term,0);

# Customer Dimension Table
SELECT ROW_NUMBER() OVER (ORDER BY (SELECT NULL)) AS CUSTOMER_KEY,Gender,Married,Dependents,
Education,Self_Employed INTO DIM_CUSTOMER
FROM ( SELECT DISTINCT Gender,Married,Dependents,Education,Self_Employed
       FROM LOAN_DATA)T;

# Property Dimension Table 
SELECT ROW_NUMBER() OVER (ORDER BY Property_Area) AS PROPERTY_KEY, Property_Area
INTO DIM_PROPERTY
FROM (SELECT DISTINCT Property_Area FROM LOAN_DATA)T;

# Credit Dimension Table
 SELECT ROW_NUMBER() OVER (ORDER BY Credit_History) AS  CREDIT_KEY, Credit_History
INTO DIM_CREDIT 
FROM ( SELECT DISTINCT Credit_History FROM LOAN_DATA)T;

# FACT TABLE
SELECT L.Loan_ID,C.CUSTOMER_KEY,P.PROPERTY_KEY,CR.CREDIT_KEY,L.ApplicantIncome,
       L.CoapplicantIncome,L.LoanAmount,L.Loan_Amount_Term,L.LOan_Status
INTO FACT_LOAN FROM LOAN_DATA L
JOIN DIM_CUSTOMER C ON L.Gender = C.Gender
                                     AND L.Married = C.Married
                                     AND L.Dependents = C.Dependents
	                      AND L.Education = C.Education
                                     AND L.Self_Employed = C.Self_Employed
JOIN DIM_PROPERTY P ON L.Property_Area = P.Property_Area
JOIN DIM_CREDIT CR ON L.Credit_History = CR.Credit_History;

# DEFAULT RATE PERCENTAGE
SELECT Loan_Status,COUNT(*)*100.0/SUM(COUNT(*)) OVER() AS PERCENTAGE FROM FACT_LOAN
GROUP BY Loan_Status;

# CREDIT RISK ANALYSIS
SELECT CR.Credit_History,COUNT(*) AS TOTAL_LOANS,
   SUM(CASE WHEN F.Loan_Status = 'N' THEN 1 ELSE 0 END) AS DEFAULTS,
   (SUM(CASE WHEN F.Loan_Status = 'N' THEN 1 ELSE 0 END) *100.0/COUNT(*)) AS DEFAULT_RATE
FROM FACT_LOAN F
JOIN DIM_CREDIT CR ON F.CREDIT_KEY = CR.CREDIT_KEY
GROUP BY CR.Credit_History;

# INCOME SEGMENTATION
SELECT CASE 
WHEN ApplicantIncome < 3000 THEN 'LOW INCOME'
WHEN ApplicantIncome BETWEEN 3000 AND 7000 THEN 'MEDIUM INCOME'   
ELSE 'HIGH INCOME'
END AS INCOME_GROUP,Loan_Status,COUNT(*) AS TOTAL
FROM FACT_LOAN
GROUP BY CASE
WHEN ApplicantIncome < 3000 THEN 'LOW INCOME'
WHEN ApplicantIncome BETWEEN 3000 AND 7000 THEN 'MEDIUM INCOME'   
ELSE 'HIGH INCOME'
END, Loan_Status;

# PROPERTY RISK
SELECT P.Property_Area,COUNT(*) AS TOTAL_LOANS,
SUM(CASE WHEN F.Loan_Status = 'N' THEN 1 ELSE 0 END) AS DEFAULTS
FROM FACT_LOAN F
JOIN DIM_PROPERTY P ON F.PROPERTY_KEY = P.PROPERTY_KEY
GROUP BY P.Property_Area;

# HIGH RISK PROFILE
SELECT TOP 10 C.Gender,C.Education,C.Self_Employed,COUNT(*) AS DEFAULT_COUNT
FROM FACT_LOAN F
JOIN DIM_CUSTOMER C ON F.CUSTOMER_KEY = C.CUSTOMER_KEY
WHERE F.Loan_Status ='N'
GROUP BY C.Gender,C.Education,C.Self_Employed
ORDER BY DEFAULT_COUNT DESC;
